<table id="csvTable">

    <thead>

    </thead>

    <tbody id="tableBody"></tbody>

</table>


<script>

    // Function to fetch and update the CSV data

    async function fetchCSVData() {

        const response = await fetch(‘test.csv');  // UPDATE FILE NAME

        const data = await response.text();

        const rows = data.split('\n').map(row => row.split(','));


        const tableHeader = document.getElementById('tableHeader');

        const tableBody = document.getElementById('tableBody');


        // Clear the existing table

        tableHeader.innerHTML = '';

        tableBody.innerHTML = '';


        // Add header from the first row

        const headers = rows[0];

        headers.forEach(header => {

            const th = document.createElement('th');

            th.textContent = header;

            tableHeader.appendChild(th);

        });


        // Add data rows

        rows.slice(1).forEach(row => {

            const tr = document.createElement('tr');

            row.forEach(cell => {

                const td = document.createElement('td');

                td.textContent = cell;

                tr.appendChild(td);

            });

            tableBody.appendChild(tr);

        });

    }


    // Fetch the data initially and set up periodic updates

    fetchCSVData();

    setInterval(fetchCSVData, 5000);  // Update every 5 seconds

</script>
